import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.nio.file.StandardOpenOption;
import java.nio.charset.StandardCharsets;

public class Program {
    public static void main(String[] args) {
        String filePath = "example.txt";

        try {
            // Write text to a file
            Files.write(Paths.get(filePath), "Hello, World!".getBytes(StandardCharsets.UTF_8));

            // Read text from a file
            String content = Files.readString(Paths.get(filePath), StandardCharsets.UTF_8);
            System.out.println("File Content: " + content);

            // Append text to a file
            Files.write(Paths.get(filePath), "\nAppended Text".getBytes(StandardCharsets.UTF_8), StandardOpenOption.APPEND);

            // Read updated content
            String updatedContent = Files.readString(Paths.get(filePath), StandardCharsets.UTF_8);
            System.out.println("Updated File Content: " + updatedContent);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
